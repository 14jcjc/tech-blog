SELECT 
  CASE 
    WHEN SUBSTR(c.postal_cd, 1, 3) BETWEEN '100' AND '209' THEN 1 
    ELSE 0

  COUNT(*) AS n_customer
FROM

INNER JOIN 
  (SELECT DISTINCT customer_id FROM receipt) r
